# ─── 1. SYSTEM REQUIREMENTS ──────────────────────────────────────────

node          >= 20.0.0     # https://nodejs.org  (or: brew install node)
npm           >= 10.0.0     # ships with Node
postgresql    >= 16.0       # brew install postgresql@16   (macOS)
                            # apt install postgresql-16    (Linux)
git           >= 2.30       # version control
bash          >= 3.2        # shell

# Verify versions:
#   node --version
#   npm --version
#   psql --version
#   git --version

# Ports used:
#   5432  PostgreSQL
#   4000  Express API
#   5173  Vite dev server


# ─── 2. START POSTGRESQL ─────────────────────────────────────────────

# macOS (Homebrew):
#   brew services start postgresql@16

# Linux:
#   sudo service postgresql start

# Verify it is running:
#   pg_isready -h localhost -p 5432
#   → localhost:5432 - accepting connections


# ─── 3. CREATE DATABASE + USER ───────────────────────────────────────

# Open the Postgres superuser shell:
#   psql postgres

# Then paste the block below at the  postgres=#  prompt:

    CREATE USER cg_user WITH PASSWORD '{your pass}';
    CREATE DATABASE community_garden OWNER cg_user;
    GRANT ALL PRIVILEGES ON DATABASE community_garden TO cg_user;
    \c community_garden
    GRANT ALL ON SCHEMA public TO cg_user;
    ALTER SCHEMA public OWNER TO cg_user;
    \q

# One-liner alternative (runs the same commands from the shell):
#   psql postgres -c "CREATE USER cg_user WITH PASSWORD 'shafiq123';" \
#                 -c "CREATE DATABASE community_garden OWNER cg_user;" \
#                 -c "GRANT ALL PRIVILEGES ON DATABASE community_garden TO cg_user;"
#   psql -d community_garden postgres -c "GRANT ALL ON SCHEMA public TO cg_user;" \
#                                     -c "ALTER SCHEMA public OWNER TO cg_user;"

# Verify the new user works:
#   psql -h localhost -U cg_user -d community_garden
#   (password: shafiq123 — then \q to quit)

# ─── 11. TROUBLESHOOTING ─────────────────────────────────────────────

# "Invalid email or password" at login
#   → DB not seeded.   cd server && npm run seed

# "connect ECONNREFUSED 127.0.0.1:5432"
#   → Postgres is not running.   brew services start postgresql@16

# "password authentication failed for user 'cg_user'"
#   → Password mismatch between DB and server/.env.
#     Reset with:  psql postgres -c "ALTER USER cg_user WITH PASSWORD 'shafiq123';"

# "relation does not exist"
#   → Migrations not run.   cd server && npm run migrate

# Port already in use (4000 or 5173)
#   → lsof -i :4000      (find the PID holding the port)
#     kill -9 <PID>

# Frontend shows "Network Error"
#   → Backend crashed. Check Terminal 1.

# npm install fails with native-module errors on Node 23
#   → Try:   rm -rf node_modules package-lock.json && npm install
